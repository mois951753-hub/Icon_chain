Shader "Custom/TerrainSimpleTest"
{
    Properties
    {
        _BaseHeightTex ("BaseHeight", 2D) = "white" {}
        _HeightScale ("HeightScale", Float) = 50.0
        _ChunkWorldX ("ChunkWorldX", Float) = 0
        _ChunkWorldZ ("ChunkWorldZ", Float) = 0
        _ChunkSize ("ChunkSize", Float) = 128
    }
    
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma target 3.0
            #include "UnityCG.cginc"
            
            sampler2D _BaseHeightTex;
            float _HeightScale;
            float _ChunkWorldX;
            float _ChunkWorldZ;
            float _ChunkSize;
            
            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };
            
            struct v2f
            {
                float4 pos : SV_POSITION;
                float3 worldPos : TEXCOORD0;
                float height : TEXCOORD1;
            };
            
            v2f vert(appdata v)
            {
                v2f o;
                
                // 方法1：直接使用局部UV采样
                float height = tex2Dlod(_BaseHeightTex, float4(v.uv, 0, 0)).r * _HeightScale;
                
                // 方法2：使用世界坐标计算全局UV（测试用）
                // float worldX = v.vertex.x + _ChunkWorldX;
                // float worldZ = v.vertex.z + _ChunkWorldZ;
                // float2 globalUV = float2(worldX / (_ChunkSize * 8.0), worldZ / (_ChunkSize * 8.0)); // 假设8个chunk
                // float height = tex2Dlod(_BaseHeightTex, float4(globalUV, 0, 0)).r * _HeightScale;
                
                // 计算世界位置
                float3 worldPos;
                worldPos.x = v.vertex.x + _ChunkWorldX;
                worldPos.z = v.vertex.z + _ChunkWorldZ;
                worldPos.y = height;
                
                // 对象空间位置（顶点本地坐标 + 高度）
                float3 objPos = float3(v.vertex.x, height, v.vertex.z);
                
                o.pos = UnityObjectToClipPos(float4(objPos, 1.0));
                o.worldPos = worldPos;
                o.height = height;
                
                return o;
            }
            
            fixed4 frag(v2f i) : SV_Target
            {
                // 显示高度（红色）和边缘（蓝色）
                float edge = 0;
                if (i.worldPos.x - _ChunkWorldX < 2.0 || 
                    i.worldPos.x - _ChunkWorldX > _ChunkSize - 2.0 ||
                    i.worldPos.z - _ChunkWorldZ < 2.0 || 
                    i.worldPos.z - _ChunkWorldZ > _ChunkSize - 2.0)
                {
                    edge = 1.0; // 边缘显示蓝色
                }
                
                float heightNormalized = i.height / _HeightScale;
                return fixed4(heightNormalized, edge, 0, 1);
            }
            ENDCG
        }
    }
    FallBack "Diffuse"
}
